36 template <typename T>
37 class mutable_compute_cell_state :

54 mutable_compute_cell_state(key_ref key, const std::unordered_set<cell> &arguments) :
55 parent(key),
56 arguments(arguments) {}

63 T value() {
64 if (stale) {
65 computed = true;
66
67 try {
68 this->silent_set(compute());
69 }
70 catch (const stop_compute_exception &) {
71 // Prevent value from being updated
72 }
73
74 stale = !this->is_active();
75 }
76
77 return parent::value();
78 }
79
85 void value(T value) {
86 reverse = true;
87 this->notify_will_update();
88
89 updating = false;
90 stale = false;
91 computed = false;
92
93 this->silent_set(value);
94
95 try {
96 batch([&, this] () {
97 reverse_compute(value);
98 });
99 }
100 catch (...) {
101 // Prevent exception from being propagated to caller
102 }
103
104 if (parent::is_batch_update()) {
105 this->add_to_batch();
106 }
107 else {
108 this->notify_update();
109 }
110
111 reverse = false;
112 }
113
114 void will_update(const key_ref &k) override {
115 if (!reverse && !updating) {
116 updating = true;
117
118 this->notify_will_update();
119 stale = true;
120 }
121 }
122
123 void update(const key_ref &k) override {
124 if (updating) {
125 this->notify_update();
126 updating = false;
127 }
128 }
129
134 void init() override {
135 cell_state::init();
136
137 try {
138 this->silent_set(this->value());
139 }
140 catch (...) {
141 // Prevent exception from being propagated to caller
142 }
143
144 for (auto arg : arguments) {
145 arg.add_observer(observer_ptr());
146 }
147 }
148
153 void pause() override {
154 cell_state::pause();
155 stale = true;
156
157 for (auto arg : arguments) {
158 arg.remove_observer(observer_ptr());
159 }
160 }
161
162 protected:
167 std::unordered_set<cell> arguments;
168
172 bool stale = true;
173
178 bool updating = false;
179
184 bool reverse = false;
185
190 bool computed = true;
191
201 virtual T compute() = 0;
202
215 virtual void reverse_compute(T value) = 0;
216
224 std::shared_ptr<observer> observer_ptr() {
225 return std::dynamic_pointer_cast<observer>(this->shared_from_this());
226 }
227 };
