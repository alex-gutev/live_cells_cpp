52 template <typename T, typename U>
53 dynamic_mutable_compute_cell_state(key_ref key, T&& compute, U&& reverse) :
54 parent(key, {}),
55 compute_fn(std::forward<T>(compute)),
56 reverse_fn(std::forward<U>(reverse)) {}
57
58 protected:
59 value_type compute() override {
60 auto t = argument_tracker::global().with_tracker([this] (auto cell) {
61 if (!this->arguments.count(cell)) {
62 this->arguments.emplace(cell);
63
64 cell.add_observer(this->observer_ptr());
65 }
66 });
67
68 return compute_fn();
69 }
70
71 void reverse_compute(value_type value) override {
72 reverse_fn(value);
73 }
74

80 R reverse_fn;
