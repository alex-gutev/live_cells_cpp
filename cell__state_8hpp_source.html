171 std::shared_ptr<S> get(key_ref k, Args... args) {
172 if (k->is_unique()) {
173 return std::make_shared<S>(k, args...);
174 }
175
176 // TODO: Ensure that dynamic_pointer_cast succeeds.
177
178 auto it = states.find(k);
179
180 if (it != states.end()) {
181 if (auto state = it->second.lock()) {
182 return std::dynamic_pointer_cast<S>(state);
183 }
184 else {
185 states.erase(it);
186 }
187 }
188
189 auto state = states.emplace(k, std::weak_ptr<S>(std::make_shared<S>(k, args...))).first->second.lock();
190 return std::dynamic_pointer_cast<S>(state);
191 }
192
198 void remove(key_ref k) {
199 if (!k->is_unique()) {
200 states.erase(k);
201 }
202 }
