169 std::shared_ptr<S> get(key_ref k, Args... args) {
170 if (k->is_unique()) {
171 return std::make_shared<S>(k, args...);
172 }
173
174 // TODO: Ensure that dynamic_pointer_cast succeeds.
175
176 auto it = states.find(k);
177
178 if (it != states.end()) {
179 if (auto state = it->second.lock()) {
180 return std::dynamic_pointer_cast<S>(state);
181 }
182 else {
183 states.erase(it);
184 }
185 }
186
187 auto state = states.emplace(k, std::weak_ptr<S>(std::make_shared<S>(k, args...))).first->second.lock();
188 return std::dynamic_pointer_cast<S>(state);
189 }
190
196 void remove(key_ref k) {
197 if (!k->is_unique()) {
198 states.erase(k);
199 }
200 }
