35 template <typename T, typename U>
36 struct eq_cell_key : key {
37 const T a;
38 const U b;
39
40 eq_cell_key(T a, U b) : a(a), b(b) {}
41
42 bool eq(const key &k) const noexcept override {
43 auto *ptr = dynamic_cast<const eq_cell_key*>(&k);
44
45 return ptr != nullptr &&
46 a.key() == ptr->a.key() &&
47 b.key() == ptr->b.key();
48 }
49
50 size_t hash() const noexcept override {
